The C Programming Language에 실려있는 매우 유서깊은 방식. 다만 요즘엔 이런식으로 적는 사람이 거의 없다(...)  

    
    
    #include <stdio.h>main(){    printf("Hello, World!\n");}

  

혹은 이렇게 하거나  

    
    
    #include <stdio.h>int main(int argc, const char * argv[]){    printf("Hello, World!\n");    return 0;}

  

또는 이렇게 하거나

  

    
    
    #include <stdio.h>int main() {    printf("Hello, World!\n");    return 0;}

  

또는 아래와 같이 해도된다

  

    
    
    #include <stdio.h>int main(void){     puts("Hello, World!");//puts는 printf와는 달리 '\n'이 자동으로 끝에 삽입된다.    return 0;}

  

위와 거의 비슷하게 이렇게도 코딩 가능하다.

  

    
    
    #include <stdio.h>int main(void){     fputs("Hello, World!\n",stdout);    return 0;}

  

또는 변태적으로(...) 이렇게 해도 되긴 된다. <del>하지만 그냥 윗 내용들 쓰는게 빠르다</del>  

    
    
    #include <stdio.h>int main(){    char str[20] = "Hello, World!"; // 문자열 생성    int lo; // for문 돌리기 위한 int형 변수    for(lo = 0; str[lo] != NULL; lo++) // char 형으로 배열(문자열 저장을 위해) 생성시 데이터 입력후 남은 공간은 NULL로 초기화 된다.    {         printf("%c", str[lo]); // 문자 하나하나 출력(...)    }    printf("\n");    return 0;}
